	placement = ROOT

	flavor = dvg_china.1.f

	event_image = {
		video = "unspecific_military_parade"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/unspecific/military_parade"

	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	duration = 3

	trigger = {
	}

	option = {
		name = dvg_china.1.a
		default_option = yes
		add_loyalists = {
			value = 0.1
			culture = cu:han
		}
		every_state_region = {
			limit = {
			OR = {
				is_homeland = cu:han
				is_homeland = cu:hakka
				is_homeland = cu:min
				is_homeland = cu:yue
				}
			}
			add_claim = ROOT
		}
		if = {
			limit = {
				country_has_state_religion = rel:taiping
			}
			trigger_event = {id = dvg_xijiao.6 }

		}

		change_infamy = -1000
		change_infamy = 25
		change_tag = CHI
		set_global_variable = china_reunified
	}
}

	}

	on_opened_soundeffect = "event:/SFX/Events/unspecific/military_parade"

	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	duration = 3

	trigger = {
		exists = c:CHI
		
	}

	option = {
		name = dvg_china.2.a
		trigger = {
			any_neighbouring_state = {
				owner = c:CHI
			}
		}
		
		c:CHI = {
			annex = ROOT
		}
	
		ai_chance = {
			base = 100
		}
	}
	option = {
		name = dvg_china.2.b
		trigger = {
			any_neighbouring_state = {
				owner = c:CHI
			}
		}
		create_diplomatic_play = {
			type = dp_unify_china

			target_country = c:CHI

			war = no

			add_war_goal = {
				holder = ROOT
				type = annex_country
			}
		}
		ai_chance = {
			base = 100
		}
	}
	option = {
		name = dvg_china.2.b

		add_loyalists = {
			value = 0.30
		}
		
		if = {
			limit = {
				any_primary_culture = {
					OR = {
					NOT = {
						has_discrimination_trait = han_chinese
					}
					NOT = {
						has_discrimination_trait = sinosphere
					}
					}
				}
			 }
			 remove_primary_culture = cu:han
		}


		ai_chance = {
			base = 1
		}
	}
}
