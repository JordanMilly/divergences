dvg_integrate_east_qq = {
	is_shown = {
		exists = c:QIN
		this = c:QIN
		root = {
			OR = {
				AND = {
					owns_entire_state_region = STATE_JINSHAN
					s:STATE_JINSHAN = {
						NOT = {
							is_homeland = cu:qingqiuan
						}
					}
				}
				AND = {
					owns_entire_state_region = STATE_YICHENG
					s:STATE_YICHENG = {
						NOT = {
							is_homeland = cu:qingqiuan
						}
					}
				}
				OR = {
					AND = {
						has_state_in_state_region = STATE_COMANCHERIE
						s:STATE_COMANCHERIE = {
							is_incorporated = yes
							NOT = {
								is_homeland = cu:qingqiuan
							}
						}
					}
					AND = {
						has_state_in_state_region = STATE_TERRAS_VERMELHAS
						s:STATE_TERRAS_VERMELHAS = {
							is_incorporated = yes
							NOT = {
								is_homeland = cu:qingqiuan
							}
						}
					}
				}
			}
		}
		NOT = {
			has_law = law_type:law_multicultural
		}
	}

    when_taken = {
		IF = {
			limit = {
				c:QIN = {
					owns_entire_state_region = STATE_JINSHAN
					}
				s:STATE_JINSHAN = {
					NOT = {
						is_homeland = cu:qingqiuan
					}
				}
			}
			s:STATE_JINSHAN = {
				add_homeland = cu:qingqiuan
			}
		}
		IF = {
			limit = {
				c:QIN = {
					owns_entire_state_region = STATE_YICHENG
					}
				s:STATE_YICHENG = {
					NOT = {
						is_homeland = cu:qingqiuan
					}
				}
			}
			s:STATE_YICHENG = {
				add_homeland = cu:qingqiuan
			}
		}
		IF = {
			limit = {
				c:QIN = {
					has_state_in_state_region = STATE_COMANCHERIE
					s:STATE_COMANCHERIE = {
						NOT = {
							is_homeland = cu:qingqiuan
						}
					}
				}
			}
			s:STATE_COMANCHERIE = {
				add_homeland = cu:qingqiuan
			}
		}
		IF = {
			limit = {
				c:QIN = {
					has_state_in_state_region = STATE_TERRAS_VERMELHAS
					s:STATE_TERRAS_VERMELHAS = {
						NOT = {
							is_homeland = cu:qingqiuan
						}
					}
				}
			}
			s:STATE_TERRAS_VERMELHAS = {
				add_homeland = cu:qingqiuan
			}
		}
    }
}
